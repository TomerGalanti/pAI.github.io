def solve(graph):
    # Public contract: return a valid dominating set.
    n = graph.num_vertices()
    deg = graph.degrees()

    # pAI-managed region
    uncovered = all_vertices(graph)
    solution = []

    while uncovered:
        v = best_greedy_vertex(graph, uncovered)
        solution.append(v)
        uncovered -= graph.closed_neighbors(v)

    solution = repair_and_prune(graph, solution)    core = degree_bucket_seed(graph, deg)
    solution = distribution_tuned_greedy(
        graph,
        seed=core,
        lookahead=2,
        prune_budget=1400,
    )

    solution = private_pace_repair(graph, solution)
    solution = local_delete_prune(graph, solution)
    # end pAI-managed region

    assert is_dominating_set(graph, solution)
    return solution
